# Reversing Conway's Game of Life

A port of [lotz84's notebook](https://lotz84.github.io/haskell-notebooks/posts/2024-10-20/).

The Game of Life runs forwards by a fixed rule. Running it *backwards* i.e finding a
board whose next generation is some target, is hard, because the rule throws
information away. The trick in the original post is to relax the boolean rule into a
**differentiable** one (`&&` becomes `*`, `||` becomes `x + y - xy`, the
neighbour comparisons become sigmoids), so a single step is a smooth function we can
push a target image back through with gradient descent.

```haskell
{-# LANGUAGE DeriveFunctor, DeriveFoldable, DeriveTraversable #-}
-- | A list zipper: a focused element with neighbours either side.
data Z a = Z [a] a [a]
  deriving (Show, Functor, Foldable, Traversable)

dup :: a -> (a, a)
dup x = (x, x)

left :: Z a -> Maybe (Z a)
left (Z [] _ _)      = Nothing
left (Z (l:ls) c rs) = Just (Z ls l (c:rs))

right :: Z a -> Maybe (Z a)
right (Z _ _ [])      = Nothing
right (Z ls c (r:rs)) = Just (Z (c:ls) r rs)

-- | Apply f until it returns Nothing, collecting the results.
iterate1 :: (a -> Maybe a) -> a -> [a]
iterate1 f = unfoldr (fmap dup . f)

instance Comonad Z where
  extract (Z _ a _) = a
  duplicate z = Z (iterate1 left z) z (iterate1 right z)
  extend f z = Z (f <$> iterate1 left z) (f z) (f <$> iterate1 right z)

toZ :: [a] -> Z a
toZ []     = error "toZ can't take an empty list."
toZ (x:xs) = Z [] x xs

unZ :: Z a -> [a]
unZ (Z ls c rs) = reverse ls ++ [c] ++ rs

zipZWith :: (a -> b -> c) -> Z a -> Z b -> Z c
zipZWith f (Z ls1 c1 rs1) (Z ls2 c2 rs2) =
  Z (zipWith f ls1 ls2) (f c1 c2) (zipWith f rs1 rs2)
```

```haskell
{-# LANGUAGE DeriveFunctor, DeriveFoldable, DeriveTraversable #-}
-- | A Game of Life board: a zipper of rows, each a zipper of cells.
newtype Z2 a = Z2 (Z (Z a))
  deriving (Show, Functor, Foldable, Traversable)

instance Comonad Z2 where
  extract (Z2 zz) = extract (extract zz)
  duplicate (Z2 zz) = fmap Z2 . Z2 . roll $ roll zz
    where
      roll zz = Z (iterate1 (mapM left) zz) zz (iterate1 (mapM right) zz)

toZ2 :: [[a]] -> Z2 a
toZ2 []     = error "toZ2 can't take an empty list."
toZ2 (x:xs) = Z2 (Z [] (toZ x) (map toZ xs))

unZ2 :: Z2 a -> [[a]]
unZ2 (Z2 (Z lzs cz rzs)) = map unZ (reverse lzs) ++ [unZ cz] ++ map unZ rzs

zipZ2With :: (a -> b -> c) -> Z2 a -> Z2 b -> Z2 c
zipZ2With f (Z2 (Z lzs1 cz1 rzs1)) (Z2 (Z lzs2 cz2 rzs2)) =
  Z2 (Z (zipWith (zipZWith f) lzs1 lzs2)
        (zipZWith f cz1 cz2)
        (zipWith (zipZWith f) rzs1 rzs2))
```

```haskell
-- | Count live cells in one row's neighbourhood (optionally the centre).
countNeighbour :: Bool -> Z Bool -> Int
countNeighbour self (Z ls c rs) = length $ filter id
  [ listToMaybe ls == Just True
  , self && c
  , listToMaybe rs == Just True
  ]

-- | Count the eight neighbours of the focused cell.
countNeighbours :: Z2 Bool -> Int
countNeighbours (Z2 (Z lzs cz rzs)) = sum
  [ maybe 0 (countNeighbour True) (listToMaybe lzs)
  , countNeighbour False cz
  , maybe 0 (countNeighbour True) (listToMaybe rzs)
  ]

-- | The classic rule for one cell: survive on 2 or 3, born on 3.
life :: Z2 Bool -> Bool
life z = (a && (n == 2 || n == 3)) || (not a && n == 3)
  where
    a = extract z
    n = countNeighbours z
```

```haskell
-- | Live cells are dark, the backdrop light.
liveColor :: Color
liveColor = rgb 46 52 64

deadColor :: Color
deadColor = rgb 236 239 244

-- | A board as a 'Picture': a light backdrop with one dark square per live cell.
boardPic :: Double -> [[Bool]] -> Picture
boardPic px rows =
  fill deadColor (rectangle (0, 0) w h) <> group
    [ translate (fromIntegral x * px, fromIntegral y * px)
        (fill liveColor (rectangle (0, 0) px px))
    | (y, row) <- zip [0 :: Int ..] rows, (x, c) <- zip [0 :: Int ..] row, c ]
  where
    ncols = case rows of { [] -> 0; (r:_) -> length r }
    w = fromIntegral ncols * px
    h = fromIntegral (length rows) * px

-- | The canvas a board of the given cell size needs.
boardCanvas :: Double -> [[Bool]] -> Canvas
boardCanvas px rows = Canvas (fromIntegral ncols * px) (fromIntegral (length rows) * px)
  where ncols = case rows of { [] -> 0; (r:_) -> length r }

-- | Show a boolean board inline, no hand-written SVG.
showBoard :: Double -> Z2 Bool -> IO ()
showBoard px b = let rows = unZ2 b in pictureOn (boardCanvas px rows) (boardPic px rows)

-- | A soft board ([0,1] values) as greyscale: each cell's darkness is its value.
softBoardPic :: Double -> [[Double]] -> Picture
softBoardPic px rows =
  fill deadColor (rectangle (0, 0) w h) <> group
    [ translate (fromIntegral x * px, fromIntegral y * px)
        (opacity v (fill liveColor (rectangle (0, 0) px px)))
    | (y, row) <- zip [0 :: Int ..] rows, (x, v) <- zip [0 :: Int ..] row, v > 0.02 ]
  where
    ncols = case rows of { [] -> 0; (r:_) -> length r }
    w = fromIntegral ncols * px
    h = fromIntegral (length rows) * px

-- | Lay several equal-width board pictures in a row, with a small gap.
rowOf :: Double -> [Picture] -> Picture
rowOf stepX pics =
  group [translate (fromIntegral i * stepX, 0) p | (i, p) <- zip [0 :: Int ..] pics]
```

```haskell
-- | Centre a small ASCII pattern ('O' = live) inside an n-by-n board.
pad :: Int -> [String] -> [[Bool]]
pad n rows = top ++ body ++ bottom
  where
    h = length rows
    w = maximum (map length rows)
    cellOf ch = ch == 'O'
    padRow r = map cellOf (r ++ replicate (w - length r) '.')
    leftPad = (n - w) `div` 2
    topPad  = (n - h) `div` 2
    body = [ replicate leftPad False ++ padRow r ++ replicate (n - leftPad - w) False | r <- rows ]
    blank = replicate n False
    top = replicate topPad blank
    bottom = replicate (n - topPad - h) blank

gliderBoard :: Z2 Bool
gliderBoard = toZ2 (pad 18 [".O.", "..O", "OOO"])

showBoard 14 gliderBoard
```

## Forward: watching it run

Evolution is one line: `extend life` lifts the single-cell rule over the whole
board through the comonad. Iterating it gives the generations. The renderer below
turns a list of boards into a single looping SVG, animating each cell's colour
through the frames with SMIL `discrete` timing, so the whole grid steps in lockstep.

```haskell
step :: Z2 Bool -> Z2 Bool
step = extend life

generations :: Z2 Bool -> [Z2 Bool]
generations = iterate step

animateLife :: Double -> Double -> Int -> Z2 Bool -> IO ()
animateLife px secs n b0 = animateWith opts secs frame
  where
    gens = map unZ2 (take n (generations b0))
    (g0:_) = gens
    fps = max 1 (round (fromIntegral n / secs) :: Int)
    opts = AnimOpts (boardCanvas px g0) fps
    frame t = boardPic px (gens !! i)
      where i = min (n - 1) (floor (t / secs * fromIntegral n) :: Int)

animateLife 14 6 24 gliderBoard
```

## Backward: a differentiable rule

To run the rule *backwards* we make it smooth. Booleans become numbers in
`[0, 1]`, and each logical operation gets a differentiable stand-in: `&&` is a
product, `||` is `x + y - xy`, negation is `1 - x`, and the neighbour-count
comparisons become sigmoids. With that, one Game of Life step is a differentiable
function, so we can ask automatic differentiation which initial board would land
on a target.

```haskell
-- | Smooth stand-ins for the boolean operators, all on values in [0, 1].
gtD :: Floating a => a -> a -> a
gtD x y = 1 / (1 + exp (-4 * (x - y)))

ltD :: Floating a => a -> a -> a
ltD x y = 1 - gtD x y

eqD :: Floating a => a -> a -> a
eqD x y = exp (-(4 * (x - y))^2 / 2)

notD :: Num a => a -> a
notD x = 1 - x

andD :: Num a => a -> a -> a
andD x y = x * y

orD :: Num a => a -> a -> a
orD x y = x + y - x * y

geqD :: Floating a => a -> a -> a
geqD x y = eqD x y `orD` gtD x y

leqD :: Floating a => a -> a -> a
leqD x y = eqD x y `orD` ltD x y
```

```haskell
-- | Sum one row's neighbourhood (optionally including the centre).
sumNeighbour :: Num a => Bool -> Z a -> a
sumNeighbour self (Z ls c rs) = sum
  [ fromMaybe 0 (listToMaybe ls)
  , if self then c else 0
  , fromMaybe 0 (listToMaybe rs)
  ]

-- | Sum the eight neighbours of the focused cell.
sumNeighbours :: Num a => Z2 a -> a
sumNeighbours (Z2 (Z lzs cz rzs)) = sum
  [ maybe 0 (sumNeighbour True) (listToMaybe lzs)
  , sumNeighbour False cz
  , maybe 0 (sumNeighbour True) (listToMaybe rzs)
  ]

-- | The differentiable rule: the same logic as 'life', smooth operators.
lifeD :: Floating a => Z2 a -> a
lifeD z = (a `andD` (geqD n 2 `andD` leqD n 3)) `orD` (notD a `andD` eqD n 3)
  where
    a = extract z
    n = sumNeighbours z
```

```haskell
-- | Map [0,1] cells onto the whole real line, so descent is unconstrained.
sigmoidD :: Floating a => a -> a
sigmoidD x = 1 / (1 + exp (-x))

-- | Mean-squared error between a boolean target and one differentiable step
-- of the (sigmoid-squashed) state. Polymorphic so AD can differentiate it.
mkLoss :: Floating a => Z2 Bool -> Z2 a -> a
mkLoss tgt state =
  sum (fmap (^2) (zipZ2With (-) target (extend lifeD (fmap sigmoidD state))))
  where target = fmap (\b -> if b then 1 else 0) tgt

-- | A random starting board of real values in [-1, 1].
randomBoard :: Int -> IO (Z2 Double)
randomBoard n = toZ2 <$> replicateM n (replicateM n (randomRIO (-1, 1)))

-- | Gradient descent: returns the recovered board (squashed to [0,1]) and the
-- loss at every epoch.
descend :: Z2 Bool -> Double -> Int -> Z2 Double -> (Z2 Double, [Double])
descend tgt lr epochs s0 = go s0 epochs []
  where
    go s 0 ls = (fmap sigmoidD s, reverse ls)
    go s n ls =
      let g  = grad (mkLoss tgt) s
          s' = zipZ2With (\x gx -> x - lr * gx) s g
          l  = mkLoss tgt s' :: Double
      in go s' (n - 1) (l : ls)
```

## Running the reversal

We make the target **reachable** by stepping a real board forward once, so a valid
preimage is guaranteed to exist. Then we throw away that board, start from noise,
and let automatic differentiation push the noise back through one differentiable
step until it lands on the target. This runs at 16×16 for a couple of thousand
epochs, which finishes in a few seconds in the live kernel.

```haskell
revSource :: Z2 Bool
revSource = toZ2 (pad 16 ["..O..", ".OOO.", "OO.OO", ".OOO.", "..O.."])

revTarget :: Z2 Bool
revTarget = step revSource
```

```haskell
(revRec, revLosses) <- do
  s0 <- randomBoard 16
  pure (descend revTarget 0.2 2000 s0)
```

## Interactive: scrub the generations

The slider below is a `Behavior` from Sabela's reactive layer. Dragging it sets a
generation and **re-runs this cell**, so Sabela recomputes the board at that step
and redraws it. Same reactive mechanism as a spreadsheet: change an input, the
dependent outputs refresh.

```haskell
display (slider "gen" 0 0 23) >>= \g ->
  showBoard 14 (generations revSource !! fromIntegral g)
```
