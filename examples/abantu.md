# Bantu Language Explorer

Explore pairwise similarity across the Bantu language family using three signals derived from open datasets:

- **Lexical**: word-form similarity on the Swadesh 200-word list (core vocabulary shared by nearly all languages)
- **Name**: string similarity between language names after stripping common Bantu noun-class prefixes (*isi-*, *chi-*, *ki-*, *lu-*, …)
- **Geographic**: great-circle proximity of the regions where each language is spoken

Parameters are fixed at the marimo notebook defaults:

| Parameter | Value |
|-----------|-------|
| Reference language | **Zulu** |
| Weight — lexical | 5 |
| Weight — name | 3 |
| Weight — geographic | 2 |
| Min shared Swadesh terms | 10 |

Data files are read from `./examples/data/` relative to Sabela's working directory.

```haskell
-- String utilities -------------------------------------------------------

stripWS :: String -> String
stripWS = reverse . dropWhile isSpace . reverse . dropWhile isSpace

splitOn :: Char -> String -> [String]
splitOn c s = case break (== c) s of
  (x, [])     -> [x]
  (x, _:rest) -> x : splitOn c rest

-- LCS-based sequence match ratio (mirrors Python difflib.SequenceMatcher.ratio)
lcsLength :: String -> String -> Int
lcsLength _ [] = 0
lcsLength [] _ = 0
lcsLength a b  = last $ foldl' step (replicate (length b + 1) 0) a
  where
    step prev x = 0 : go 0 (zip3 b prev (tail prev))
      where
        go _    []                        = []
        go left ((y, diag, above) : rest) =
          let cur = if x == y then diag + 1 else max left above
          in cur : go cur rest

seqMatchRatio :: String -> String -> Double
seqMatchRatio a b
  | null a && null b = 1.0
  | null a || null b = 0.0
  | otherwise        = 2.0 * fromIntegral (lcsLength a b)
                           / fromIntegral (length a + length b)

-- Great-circle distance in km
haversine :: (Double, Double) -> (Double, Double) -> Double
haversine (lat1, lon1) (lat2, lon2) =
  let r    = 6371.0
      dlat = (lat2 - lat1) * pi / 180.0
      dlon = (lon2 - lon1) * pi / 180.0
      a    = sin (dlat/2)^2
               + cos (lat1*pi/180) * cos (lat2*pi/180) * sin (dlon/2)^2
      c    = 2 * atan2 (sqrt a) (sqrt (1 - a))
  in r * c

geoScore :: Double -> Double
geoScore d = 1.0 / (1.0 + d / 500.0)

-- RdYlGn colorscale: 0 → red, 0.5 → yellow, 1 → green
interpolateColor :: Double -> (Int, Int, Int)
interpolateColor t
  | t <= 0.5  =
      let s = t * 2
      in ( 215
         , round (25  + (255 - 25)  * s)
         , round (28  + (191 - 28)  * s)
         )
  | otherwise =
      let s = (t - 0.5) * 2
      in ( round (255 + (26  - 255) * s)
         , round (255 + (150 - 255) * s)
         , round (191 + (65  - 191) * s)
         )

toHex :: Int -> String
toHex n = let h = showHex n "" in if length h < 2 then '0':h else h

colorToHex :: (Int, Int, Int) -> String
colorToHex (r, g, b) = '#' : toHex r ++ toHex g ++ toHex b
```

```haskell
import Data.Maybe

loadSwadesh :: IO (Map.Map String (Map.Map String [String]))
loadSwadesh = do
  df <- readCsvWithOpts defaultReadOptions { typeSpec = NoInference }
            "../abantu/datasets/swadesh_list.csv"
  let allCols         = DF.columnNames df
      featCols        = filter (/= "Language") allCols
      swadeshLangVals = DF.columnAsList (DF.col @Text "Language") df
      featData        = [ ( T.unpack fc
                          , V.fromList (map (fromMaybe "") $ DF.columnAsList (DF.col @(Maybe Text) fc) df)
                          )
                        | fc <- featCols
                        ]
      swadeshEntries  = catMaybes
        [ let langStr = T.unpack lang
          in if null langStr || '"' `elem` langStr then Nothing
             else
               let feats = Map.fromList
                     [ (fcName, variants)
                     | (fcName, colVec) <- featData
                     , i < V.length colVec
                     , let rawVal   = T.unpack (colVec V.! i)
                           variants = filter validWord
                                    $ map (stripWS . map toLower)
                                    $ splitOn ';' rawVal
                     , not (null variants)
                     ]
               in if Map.null feats then Nothing
                  else Just (langStr, feats)
        | (i, lang) <- zip [0..] swadeshLangVals
        ]
      deduped = Map.fromList (reverse swadeshEntries)
  putStrLn $ "Loaded " ++ show (Map.size deduped) ++ " languages from Swadesh list"
  return deduped
  where
    validWord w = not (null w) && w `notElem` ["--","nan","n/a","-"]

swadeshData <- loadSwadesh
```

```haskell
parseSpeakers :: String -> Int
parseSpeakers s =
  let digits = filter (/= ',') $ takeWhile (\c -> c == ',' || (c >= '0' && c <= '9')) (stripWS s)
  in case reads digits of { [(n, _)] -> n; _ -> 0 }

lookupCountryCoords :: String -> (Double, Double)
lookupCountryCoords countries =
  let cs    = map (map toLower . stripWS) (splitOn ',' countries)
      found = mapMaybe (`Map.lookup` countryCentroids) cs
  in if null found
     then (0/0, 0/0)
     else ( sum (map fst found) / fromIntegral (length found)
          , sum (map snd found) / fromIntegral (length found) )
```

```haskell
-- Bantu noun-class prefix stripping for name similarity
bantuPrefixes :: [String]
bantuPrefixes =
  [ "oshi", "umu", "isi", "chi", "aki", "eki", "olu", "ebu", "omu"
  , "ny", "ku", "lu", "ru", "si", "se", "di", "ti", "ki", "gu", "bu"
  , "m", "n", "u", "i", "o" ]

stripLangPrefix :: String -> String
stripLangPrefix name =
  let low = map toLower name
  in case filter (`isPrefixOf` low) bantuPrefixes of
       (p:_) -> drop (length p) low
       []    -> low

-- Lexical similarity for a single language pair
lexPairSim :: Map.Map String (Map.Map String [String]) -> String -> String -> Double
lexPairSim swadesh l1 l2 =
  let f1     = fromMaybe Map.empty (Map.lookup l1 swadesh)
      f2     = fromMaybe Map.empty (Map.lookup l2 swadesh)
      shared = Map.intersectionWith (,) f1 f2
      scores = [ maximum [ seqMatchRatio w1 w2 | w1 <- ws1, w2 <- ws2 ]
               | (ws1, ws2) <- Map.elems shared ]
  in if length scores < 10 then 0/0
     else sum scores / fromIntegral (length scores)

-- Geographic similarity for a single language pair
geoPairSim :: (Double, Double) -> (Double, Double) -> Double
geoPairSim (lat1, lon1) (lat2, lon2)
  | isNaN lat1 || isNaN lat2 = 0/0
  | otherwise                = geoScore (haversine (lat1, lon1) (lat2, lon2))

-- Build all three similarity matrices
buildMatrices :: [String] -> IO ([[Double]], [[Double]], [[Double]])
buildMatrices ls = do
  let n      = length ls
      lsVec  = V.fromList ls
      get i  = lsVec V.! i
      stems  = V.map stripLangPrefix lsVec
      coords = V.map (\l -> fromMaybe (0/0, 0/0) (Map.lookup l langCoords)) lsVec

      lexMat  = [ [ lexPairSim swadeshData (get i) (get j) | j <- [0..n-1] ]
                | i <- [0..n-1] ]
      nameMat = [ [ seqMatchRatio (stems  V.! i) (stems  V.! j) | j <- [0..n-1] ]
                | i <- [0..n-1] ]
      geoMat  = [ [ geoPairSim   (coords V.! i) (coords V.! j) | j <- [0..n-1] ]
                | i <- [0..n-1] ]
  putStrLn $ "Building matrices for " ++ show n ++ " languages (lexical may take ~30s)…"
  let _ = length (concat lexMat)   -- force evaluation
  putStrLn "Done."
  return (lexMat, nameMat, geoMat)

langs :: [String]
langs = Map.keys swadeshData

(lexMat, nameMat, geoMat) <- buildMatrices langs
```

```haskell
-- Weighted combination: lex=5, name=3, geo=2; NaN components excluded
combineSim :: Double -> Double -> Double -> Double
combineSim lx nm gm =
  let parts = catMaybes
        [ if isNaN lx then Nothing else Just (5.0, lx)
        , if isNaN nm then Nothing else Just (3.0, nm)
        , if isNaN gm then Nothing else Just (2.0, gm)
        ]
      wSum = sum (map fst parts)
      vSum = sum (map (\(w,v) -> w*v) parts)
  in if null parts then 0/0 else vSum / wSum

langsV :: V.Vector String
langsV = V.fromList langs

combinedSim :: Map.Map (String, String) Double
combinedSim = Map.fromList
  [ ((l1, l2), combineSim lx nm gm)
  | i <- [0 .. length langs - 1]
  , j <- [0 .. length langs - 1]
  , i /= j
  , let l1 = langsV V.! i
        l2 = langsV V.! j
        lx = lexMat  !! i !! j
        nm = nameMat !! i !! j
        gm = geoMat  !! i !! j
  ]

putStrLn $ "Combined similarity computed for " ++ show (Map.size combinedSim) ++ " pairs"
```

```haskell
-- Geographic scatter map: circle per language, coloured by similarity to Zulu
let lonMin = -20.0; lonMax = 55.0
    latMin = -35.0; latMax = 15.0
    mapW   = 620.0; mapH  = 400.0
    mL     = 10.0;  mT   = 32.0

    lonToX lon = (lon - lonMin) / (lonMax - lonMin) * mapW + mL
    latToY lat = (1.0 - (lat - latMin) / (latMax - latMin)) * mapH + mT

    withCoords =
      [ (l, lat, lon, sim)
      | l <- langs
      , Just (lat, lon) <- [Map.lookup l langCoords]
      , not (isNaN lat)
      , let sim = fromMaybe (0/0) (Map.lookup ("Zulu", l) combinedSim)
      ]

    mkCircle (l, lat, lon, sim) =
      let cx   = lonToX lon
          cy   = latToY lat
          sp   = fromMaybe 0 (Map.lookup l langSpeakers)
          r    = max 3 (min 12 (logBase 10 (fromIntegral sp + 1) * 2.5))
          fill
            | l == "Zulu"  = "#1a5276"
            | isNaN sim    = "#aaa"
            | otherwise    = colorToHex (interpolateColor sim)
          lbl  = showFFloat (Just 1) cx ""
      in concat
           [ "<circle cx='", showFFloat (Just 1) cx ""
           , "' cy='",       showFFloat (Just 1) cy ""
           , "' r='",        showFFloat (Just 1) r ""
           , "' fill='", fill
           , "' stroke='white' stroke-width='0.4' opacity='0.88'/>"
           , "<text x='", showFFloat (Just 1) (cx + r + 1) ""
           , "' y='",     showFFloat (Just 1) (cy + 3) ""
           , "' font-size='5' fill='#222'>", l, "</text>"
           ]

    circles = concatMap mkCircle withCoords

    -- Colour legend strip
    legX  = mL; legY = mT + mapH + 18
    legW  = 200; legH = 10

    legendRects = concatMap (\k ->
      let t    = fromIntegral k / fromIntegral (legW :: Int)
          fill = colorToHex (interpolateColor t)
          x    = legX + fromIntegral k
      in concat
           [ "<rect x='", showFFloat (Just 1) x ""
           , "' y='",     showFFloat (Just 1) legY ""
           , "' width='1' height='", show legH, "' fill='", fill, "'/>"
           ]
      ) [0 .. legW - 1 :: Int]

    legendLabels = concat
      [ "<text x='", showFFloat (Just 1) legX ""
      , "' y='",     showFFloat (Just 1) (legY - 3) ""
      , "' font-size='7' fill='#555'>Similarity to Zulu:</text>"
      , "<text x='", showFFloat (Just 1) legX ""
      , "' y='",     showFFloat (Just 1) (legY + legH + 9) ""
      , "' font-size='7' fill='#333'>0%</text>"
      , "<text x='", showFFloat (Just 1) (legX + fromIntegral legW / 2) ""
      , "' y='",     showFFloat (Just 1) (legY + legH + 9) ""
      , "' font-size='7' text-anchor='middle' fill='#333'>50%</text>"
      , "<text x='", showFFloat (Just 1) (legX + fromIntegral legW) ""
      , "' y='",     showFFloat (Just 1) (legY + legH + 9) ""
      , "' font-size='7' text-anchor='end' fill='#333'>100%</text>"
      ]

    svgTotalH = mT + mapH + 50
    geoTitle  = concat
      [ "<text x='", showFFloat (Just 1) (mL + mapW / 2) ""
      , "' y='22' font-size='12' text-anchor='middle'"
      , " font-weight='bold' fill='#222'>"
      , "Geographic Distribution — Similarity to Zulu</text>"
      ]

      [ "<svg xmlns='http://www.w3.org/2000/svg'"

      , geoTitle

      , legendRects, legendLabels
```
